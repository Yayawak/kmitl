\section*{4.3.42
    \textbf{
        The graph of function $y = f(x)$ is shown. At which point(s) are the following true?
    }
}

\par
Analysis :
% \enumerate{
\itemize{
    \item from graph y is on above of x-axis so $y > 0$
    \item $\frac{dy}{dx}$
        \subitem at $(x,y) = B \quad \frac{dy}{dx} > 0$
        \subitem at $(x,y) = A,E \quad \frac{dy}{dx} < 0$
        \subitem at $(x,y) = C \quad \frac{dy}{dx} = 0$
}

\subsection*{A}
    $ \frac{dy}{dx} $ and $\frac{d^2y}{dx^2}$ are both positive
    Ans : B, C
\subsection*{B}
    $ \frac{dy}{dx} $ and $\frac{d^2y}{dx^2}$ are both negative
    Ans : A, E
\subsection*{C}
    $ \frac{dy}{dx} $ is negative but $\frac{d^2y}{dx^2}$ is positive
    Ans : D
