    $
    % \lim_{x \to {-2}} \frac{x^3 + 8}{x + 2}
    \underset{x \to {-2}}{\lim} \big{\frac{x^3 + 8}{x + 2}
    }
    % \limits_{x \to {-2}} \frac{x^3 + 8}{x + 2}

}
\\ \\ \\

\text{let}
\begin{align}
\begin{split}
    y &= \frac{x^3 + 8}{x + 2} \\
 &= \frac{x^3 + 2^3}{x + 2} \\
 &= \frac{(x + 2)(x^2 - 2x + 4)}{x + 2} \\
 &= (x^2 - 2x + 4) \\
\notag
\end{split}
\end{align}

\begin{align}
\begin{split}
    \underset{x \to {-2}}{\lim} \big{\frac{x^3 + 8}{x + 2}}
     &= (-2)^2 - 2(-2) + 4 \\
     &= 4 + 4 + 4 \\
     &= 12
\notag
\end{split}
\end{align}
